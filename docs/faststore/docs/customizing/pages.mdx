## Styling a section

All store content is centered by default using the `[data-fs-content]` with our `@mixin layout-content`. To add any customization, as in the examples below, you can add it to your `custom-theme.scss` file inside the `.section` class using the desired `data-attribute`.

<Tabs items={['Tiles', 'BannerText', 'Code']} defaultIndex="0">
  <Tab>
    <OverviewSection direction="column" containerStyle={{ padding: '0' }}>
      <section class="section layout__section | nx-section-layout">
        <TilesTemplate/>
      </section>
    </OverviewSection>
  </Tab>
  <Tab>
    <OverviewSection direction="column" containerStyle={{ padding: '0' }}>
      <section class="section layout__section | nx-section-layout">
        <BannerText colorVariant="light">
          <BannerTextContent
            title="Explore more about our amazing products and discover a world of possibilities."
            caption="All the amazing products from the brands we partner with."
            link="/"
            linkText="See more"
          />
        </BannerText>
      </section>
    </OverviewSection>
  </Tab>
  <Tab>
    ```scss
    // This is already set as default
    [data-fs-content] { @include layout-content; }
    ```
  </Tab>
</Tabs>

### Content centered + Colored bg

To display a **colored background** with **full width** and maintain the **content centered**, add `@include layout-colored` to your desired section.

<Tabs items={['Tiles', 'BannerText', 'Code']} defaultIndex="0">
  <Tab>
    <OverviewSection containerStyle={{ padding: '0' }}>
      <section class="section layout__section | nx-section-layout nx-section-layout-colored">
        <TilesTemplate/>
      </section>
    </OverviewSection>
  </Tab>
  <Tab>
    <OverviewSection direction="column" containerStyle={{ padding: '0' }}>
      <section class="section layout__section | nx-section-layout nx-section-layout-colored">
        <BannerText colorVariant="light">
          <BannerTextContent
            title="Explore more about our amazing products and discover a world of possibilities."
            caption="All the amazing products from the brands we partner with."
            link="/"
            linkText="See more"
          />
        </BannerText>
      </section>
    </OverviewSection>
  </Tab>
  <Tab>
    ```scss
    // Add this to your custom-theme.scss
    .section {
      [data-fs-tiles] {
        @include layout-colored(var(--fs-color-main-2));
      }

      [data-fs-banner-text] {
        @include layout-colored(var(--fs-color-main-2));
      }
    }
    ```

  </Tab>
</Tabs>

### Content centered + Colored bg + Section centered

To add a **colored background** and **centering** the `.section`, add `@include layout-colored` and `@include layout-content` to your desired section.

<Tabs items={['Tiles', 'BannerText', 'Code']} defaultIndex="0">
  <Tab>
    <OverviewSection containerStyle={{ padding: '0' }}>
      <section class="section layout__section layout__content | nx-section-layout nx-section-layout-colored">
        <TilesTemplate/>
      </section>
    </OverviewSection>
  </Tab>
  <Tab>
    <OverviewSection direction="column" containerStyle={{ padding: '0' }}>
      <section class="section layout__section layout__content | nx-section-layout nx-section-layout-colored">
        <BannerText colorVariant="light">
          <BannerTextContent
            title="Explore more about our amazing products and discover a world of possibilities."
            caption="All the amazing products from the brands we partner with."
            link="/"
            linkText="See more"
          />
        </BannerText>
      </section>
    </OverviewSection>
  </Tab>
  <Tab>
    ```scss
    // Add this to your custom-theme.scss
    .section {
      [data-fs-tiles] {
        @include layout-colored(var(--fs-color-main-2));
        @include layout-content;
      }

      [data-fs-banner-text] {
        @include layout-colored(var(--fs-color-main-2));
        @include layout-content;
      }
    }
    ```

  </Tab>
</Tabs>

### Full width layout

To use your store section's content with a **full-width layout** you'll need to change its content to include the `@mixin layout-content-full`.

>ℹ️ Our [Alert](/docs/faststore/molecules/alert) component is the only section that comes with **layout-content-full** by default.

<Tabs items={['Tiles', 'BannerText', 'Code']} defaultIndex="0">
  <Tab>
    <OverviewSection containerStyle={{ padding: '0' }}>
      <section class="section layout__section | nx-section-layout nx-section-layout-full">
        <TilesTemplate/>
      </section>
    </OverviewSection>
  </Tab>
  <Tab>
    <OverviewSection containerStyle={{ padding: '0' }}>
      <section class="section layout__section | nx-section-layout nx-section-layout-full">
        <BannerText colorVariant="light">
          <BannerTextContent
            title="Explore more about our amazing products and discover a world of possibilities."
            caption="All the amazing products from the brands we partner with."
            link="/"
            linkText="See more"
          />
        </BannerText>
      </section>
    </OverviewSection>
  </Tab>
  <Tab>
    ```scss
    // Add this to your custom-theme.scss
    .section {
      [data-fs-tiles] [data-fs-content="tiles"] { @include layout-content-full; }
      [data-fs-banner-text] [data-fs-content="banner"] { @include layout-content-full; }
    }
    ```
  </Tab>
</Tabs>
